Dãy con tăng & Dãy con chung dài nhất 📈

📈 LIS — Longest Increasing Subsequence

Tìm dãy con tăng dài nhất (không cần liên tiếp)
# O(n²)
def lis(a):
    n = len(a)
    dp = [1] * n  # dp[i] = LIS kết thúc tại i
    for i in range(1, n):
        for j in range(i):
            if a[j] < a[i]:
                dp[i] = max(dp[i], dp[j] + 1)
    return max(dp)

a = [10, 9, 2, 5, 3, 7, 101, 18]
print(lis(a))  # 4: [2, 3, 7, 101]

⚡ LIS O(n log n)

from bisect import bisect_left

def lis_fast(a):
    tails = []
    for x in a:
        pos = bisect_left(tails, x)
        if pos == len(tails):
            tails.append(x)
        else:
            tails[pos] = x
    return len(tails)

a = [10, 9, 2, 5, 3, 7, 101, 18]
print(lis_fast(a))  # 4
✅ Binary search + DP → O(n log n)!

🔗 LCS — Longest Common Subsequence

Tìm dãy con chung dài nhất của 2 chuỗi
def lcs(s1, s2):
    m, n = len(s1), len(s2)
    dp = [[0] * (n+1) for _ in range(m+1)]
    for i in range(1, m+1):
        for j in range(1, n+1):
            if s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])
    return dp[m][n]

print(lcs("ABCBDAB", "BDCAB"))  # 4: "BCAB"

📝 Edit Distance

# Số thao tác tối thiểu biến s1 → s2 (thêm/xóa/sửa)
def edit_distance(s1, s2):
    m, n = len(s1), len(s2)
    dp = [[0]*(n+1) for _ in range(m+1)]
    for i in range(m+1): dp[i][0] = i
    for j in range(n+1): dp[0][j] = j
    for i in range(1, m+1):
        for j in range(1, n+1):
            if s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1]
            else:
                dp[i][j] = 1 + min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1])
    return dp[m][n]

print(edit_distance("kitten", "sitting"))  # 3
