🔍 Linear vs Binary Search

Linear: O(n)

# Duyệt từng phần tử
def linear(a, x):
    for i in range(len(a)):
        if a[i] == x:
            return i
    return -1

Binary: O(log n)

# Chia đôi liên tục
def binary(a, x):
    lo, hi = 0, len(a)-1
    while lo <= hi:
        mid = (lo+hi)//2
        if a[mid] == x: return mid
        elif a[mid] < x: lo = mid+1
        else: hi = mid-1
    return -1
⚠️ Binary search yêu cầu mảng đã sắp xếp!

🐍 Module bisect

from bisect import bisect_left, bisect_right

a = [1, 3, 3, 5, 7, 9]

# Tìm vị trí chèn (bên trái)
print(bisect_left(a, 3))   # 1

# Tìm vị trí chèn (bên phải)
print(bisect_right(a, 3))  # 3

# Đếm số lần xuất hiện của x
def count(a, x):
    return bisect_right(a, x) - bisect_left(a, x)

print(count(a, 3))  # 2

🏆 Binary Search trên đáp án

📋 Cho dây thừng dài L, cắt thành K đoạn bằng nhau. Tìm độ dài lớn nhất mỗi đoạn.
def check(mid, ropes, k):
    total = sum(r // mid for r in ropes)
    return total >= k

lo, hi = 1, max(ropes)
ans = 0
while lo <= hi:
    mid = (lo + hi) // 2
    if check(mid, ropes, k):
        ans = mid
        lo = mid + 1
    else:
        hi = mid - 1
print(ans)
