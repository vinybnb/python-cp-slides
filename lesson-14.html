🏆 Bài kinh điển: Dấu ngoặc hợp lệ

def is_valid(s):
    stack = []
    pairs = {')': '(', ']': '[', '}': '{'}
    for ch in s:
        if ch in '([{':
            stack.append(ch)
        elif ch in ')]}':
            if not stack or stack[-1] != pairs[ch]:
                return False
            stack.pop()
    return len(stack) == 0

print(is_valid("({[]})"))   # True
print(is_valid("([)]"))     # False

📊 Monotonic Stack

📋 Cho mảng, tìm phần tử lớn hơn gần nhất bên phải mỗi phần tử
def next_greater(a):
    n = len(a)
    result = [-1] * n
    stack = []  # Lưu index
    for i in range(n):
        while stack and a[stack[-1]] < a[i]:
            result[stack.pop()] = a[i]
        stack.append(i)
    return result

a = [4, 5, 2, 10, 8]
print(next_greater(a))  # [5, 10, 10, -1, -1]
✅ O(n) thay vì O(n²) — mỗi phần tử push/pop đúng 1 lần!
