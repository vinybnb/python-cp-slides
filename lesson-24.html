🔢 Sàng Eratosthenes

Tìm tất cả số nguyên tố ≤ N — O(N log log N)
def sieve(n):
    is_prime = [True] * (n + 1)
    is_prime[0] = is_prime[1] = False
    for i in range(2, int(n**0.5) + 1):
        if is_prime[i]:
            for j in range(i*i, n + 1, i):
                is_prime[j] = False
    return [i for i in range(2, n + 1) if is_prime[i]]

primes = sieve(100)
print(primes)  # [2, 3, 5, 7, 11, 13, ...]

🔐 Phép chia dư (Modular Arithmetic)

MOD = 10**9 + 7

# Tính toán với mod — tránh tràn số
(a + b) % MOD
(a * b) % MOD
# KHÔNG: (a / b) % MOD → dùng nghịch đảo modular!

# Lũy thừa nhanh modular
def power_mod(a, n, mod):
    result = 1
    a %= mod
    while n > 0:
        if n % 2 == 1:
            result = result * a % mod
        a = a * a % mod
        n //= 2
    return result

# Nghịch đảo modular: a^(-1) mod p = a^(p-2) mod p
def mod_inverse(a, mod):
    return power_mod(a, mod - 2, mod)

🔢 Phân tích thừa số nguyên tố

def factorize(n):
    factors = {}
    d = 2
    while d * d <= n:
        while n % d == 0:
            factors[d] = factors.get(d, 0) + 1
            n //= d
        d += 1
    if n > 1:
        factors[n] = factors.get(n, 0) + 1
    return factors

print(factorize(360))  # {2: 3, 3: 2, 5: 1}

🧮 Tổ hợp C(n, k) mod p

MOD = 10**9 + 7

# Tiền xử lý giai thừa
MAX = 200001
fact = [1] * MAX
for i in range(1, MAX):
    fact[i] = fact[i-1] * i % MOD

def C(n, k):
    if k < 0 or k > n: return 0
    return fact[n] * mod_inverse(fact[k], MOD) % MOD \
                    * mod_inverse(fact[n-k], MOD) % MOD

print(C(10, 3))  # 120
