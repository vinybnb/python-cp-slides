🔍 Cách xác định Big O

# O(1) — không phụ thuộc n
x = a + b

# O(n) — 1 vòng lặp
for i in range(n): ...

# O(n²) — 2 vòng lặp lồng nhau
for i in range(n):
    for j in range(n): ...

# O(n log n) — sort
a.sort()

# O(log n) — chia đôi
while n > 0: n //= 2
