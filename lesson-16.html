🏆 Bài 1: Đổi tiền xu

Cho các mệnh giá [1, 5, 10, 25]. Đổi N đồng với ít xu nhất.
def min_coins(n):
    coins = [25, 10, 5, 1]
    count = 0
    for c in coins:
        count += n // c
        n %= c
    return count

print(min_coins(41))  # 25+10+5+1 = 4 xu
✅ Greedy đúng vì mệnh giá chia hết cho nhau!

🏆 Bài 2: Activity Selection

N hoạt động [start, end]. Chọn nhiều hoạt động nhất không trùng.
def max_activities(activities):
    # Greedy: sort theo thời gian kết thúc
    activities.sort(key=lambda x: x[1])
    count = 1
    last_end = activities[0][1]
    for i in range(1, len(activities)):
        if activities[i][0] >= last_end:
            count += 1
            last_end = activities[i][1]
    return count

acts = [(1,3), (2,5), (3,4), (5,7), (6,8)]
print(max_activities(acts))  # 3: (1,3),(3,4),(5,7)

🏆 Bài 3: Fractional Knapsack

# Túi chứa W kg. N vật (weight, value). Có thể cắt vật.
def fractional_knapsack(W, items):
    # Sort theo value/weight giảm dần
    items.sort(key=lambda x: x[1]/x[0], reverse=True)
    total = 0
    for w, v in items:
        if W >= w:
            total += v
            W -= w
        else:
            total += v * (W / w)
            break
    return total
