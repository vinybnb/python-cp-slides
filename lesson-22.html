🔍 DFS — Đi sâu trước

import sys
sys.setrecursionlimit(200000)

def dfs(u, adj, visited):
    visited[u] = True
    print(u, end=' ')
    for v in adj[u]:
        if not visited[v]:
            dfs(v, adj, visited)

# Hoặc dùng stack (không đệ quy)
def dfs_stack(start, adj, n):
    visited = [False] * (n + 1)
    stack = [start]
    while stack:
        u = stack.pop()
        if visited[u]: continue
        visited[u] = True
        for v in adj[u]:
            if not visited[v]:
                stack.append(v)

🏝️ Đếm thành phần liên thông

def count_components(n, adj):
    visited = [False] * (n + 1)
    count = 0
    for u in range(1, n + 1):
        if not visited[u]:
            dfs(u, adj, visited)
            count += 1
    return count

🔄 Phát hiện chu trình

# Đồ thị vô hướng: có chu trình khi gặp đỉnh đã thăm
def has_cycle(n, adj):
    visited = [False] * (n + 1)
    def dfs(u, parent):
        visited[u] = True
        for v in adj[u]:
            if not visited[v]:
                if dfs(v, u): return True
            elif v != parent:
                return True  # Chu trình!
        return False
    for u in range(1, n+1):
        if not visited[u]:
            if dfs(u, -1): return True
    return False
