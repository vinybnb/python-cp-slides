📦 Tạo hàm cơ bản

def chao(ten):
    print(f"Xin chào {ten}! 👋")

# Gọi hàm
chao("Minh")    # Xin chào Minh! 👋
chao("Lan")     # Xin chào Lan! 👋
📌def = define (định nghĩa hàm)
📌ten = tham số (parameter)

↩️ Hàm có return

def tong(a, b):
    return a + b

def is_prime(n):
    if n < 2: return False
    for i in range(2, int(n**0.5)+1):
        if n % i == 0: return False
    return True

# Sử dụng
print(tong(3, 5))       # 8
print(is_prime(17))      # True
print(is_prime(4))       # False
📌 return trả về kết quả — hàm dừng ngay tại đó

🏆 Bài CP: Đếm nguyên tố trong đoạn

def is_prime(n):
    if n < 2: return False
    for i in range(2, int(n**0.5)+1):
        if n % i == 0: return False
    return True

a, b = map(int, input().split())
dem = 0
for x in range(a, b+1):
    if is_prime(x):    # Gọi hàm!
        dem += 1
print(dem)
✅ Code sạch, dễ đọc, dễ debug nhờ tách hàm!

🔢 Hàm hữu ích cho CP

def gcd(a, b):          # Ước chung lớn nhất
    while b: a, b = b, a % b
    return a

def lcm(a, b):          # Bội chung nhỏ nhất
    return a * b // gcd(a, b)

def count_digits(n):    # Đếm chữ số
    dem = 0
    while n > 0:
        dem += 1
        n //= 10
    return dem
